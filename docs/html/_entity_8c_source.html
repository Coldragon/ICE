175 void ICE_Entity_MovePos(ICE_Entity * entity, ICE_Vect pos, ICE_Float r)
176 {
177 #ifdef _POLAR_MOVEMENT_TYPE_1
178 
179  // Check if it's a new movement
180  if (!entity->already_moved_polar ||
181  entity->x_polar_destination_move != pos.x ||
182  entity->y_polar_destination_move != pos.y
183  )
184  {
185  // Calculate the movement
186  ICE_Float xdif = pos.x - entity->x;
187  ICE_Float ydif = pos.y - entity->y;
188  ICE_Float angle = atan2(ydif, xdif);
189  entity->x_polar_shift_move = cos(angle);
190  entity->y_polar_shift_move = sin(angle);
191  entity->polar_distance_r_r = xdif * xdif + ydif * ydif;
192 
193  // Save for later check
194  entity->already_moved_polar = 1;
195  entity->r_polar_destination_move = r;
196  entity->x_polar_destination_move = pos.x;
197  entity->y_polar_destination_move = pos.y;
198  }
199 
200  // Movement
201  entity->x += r * entity->x_polar_shift_move;
202  entity->y += r * entity->y_polar_shift_move;
203 
204  // Check if is close to destination
205  if (entity->polar_distance_r_r < r)
206  {
207  entity->x = pos.x;
208  entity->y = pos.y;
209  }
210 
211 #endif
212 
213 #ifdef _POLAR_MOVEMENT_TYPE_2
214 
215  float xdif = pos.x - entity->x; float ydif = pos.y - entity->y;
216  float angle = atan2(ydif, xdif);
217  float distance_r_r = xdif * xdif + ydif * ydif;
218  entity->x += r * cos(angle); entity->y += r * sin(angle);
219  if (distance_r_r < r)

221  entity->x = pos.x;
